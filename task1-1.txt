    <p id="payMsg"></p>
  </div>

  <script>
    let userEmail = "";

    function register() {
      let name = document.getElementById("name").value;
      let email = document.getElementById("email").value;

      if (name === "" || email === "") {
        alert("Enter name and email");
        return;
      }

      userEmail = email;
      document.getElementById("msg").innerText = "Registered! Now login.";
      document.getElementById("login").style.display = "block";
    }

    function login() {
      let e = document.getElementById("loginEmail").value;

      if (e === userEmail) {
        document.getElementById("loginMsg").innerText = "Login success!";
        document.getElementById("payment").style.display = "block";
      } else {
        document.getElementById("loginMsg").innerText = "Invalid email!";
      }
    }

    function pay() {
      let amt = document.getElementById("amt").value;

      if (amt === "" || amt <= 0) {
        alert("Enter valid amount");
        return;
      }

      document.getElementById("payMsg").innerText = "Paid â‚¹" + amt + " successfully!";
    }
  </script>
