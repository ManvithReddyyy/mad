package com.example;

import java.io.*;
import javax.servlet.*;
import javax.servlet.http.*;

public class ConfigContextServlet extends HttpServlet {
    protected void doGet(HttpServletRequest request, HttpServletResponse response) 
            throws ServletException, IOException {
        
        response.setContentType("text/html");
        PrintWriter out = response.getWriter();
        
        // Access Servlet Config parameter
        ServletConfig config = getServletConfig();
        String author = config.getInitParameter("author");
        
        // Access Servlet Context parameter
        ServletContext context = getServletContext();
        String company = context.getInitParameter("company");
        
        // Display values
        out.println("<html><body>");
        out.println("<h2>Servlet Config and Context Parameters</h2>");
        out.println("Author (Servlet Config): " + author + "<br>");
        out.println("Company (Servlet Context): " + company + "<br>");
        out.println("</body></html>");
    }
    
    protected void doPost(HttpServletRequest request, HttpServletResponse response) 
            throws ServletException, IOException {
        doGet(request, response);
    }
}
