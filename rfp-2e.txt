package com.example.java;

import jakarta.servlet.ServletException;
import jakarta.servlet.annotation.WebServlet;
import jakarta.servlet.http.HttpServlet;
import jakarta.servlet.http.HttpServletRequest;
import jakarta.servlet.http.HttpServletResponse;
import java.io.IOException;
import java.io.PrintWriter;

/**
 * Servlet implementation class AddServlet
 */
@WebServlet("/AddServlet")
public class AddServlet extends HttpServlet {
    private static final long serialVersionUID = 1L;

    public AddServlet() {
        super();
    }

    // Handles both GET and POST requests
    protected void doGet(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {

        // Get parameters from URL or form
        String a = request.getParameter("num1");
        String b = request.getParameter("num2");

        double x = 0, y = 0;
        try {
            x = Double.parseDouble(a);
            y = Double.parseDouble(b);
        } catch (Exception e) {
        }
        double sum = x + y;
        response.setContentType("text/html");
        PrintWriter out = response.getWriter();

        out.println("<html><body>");
        out.println("<h2>Sum of " + x + " and " + y + " is: " + sum + "</h2>");
        out.println("<a href='index.html'>Go Back</a>");
        out.println("</body></html>");
    }

    // Forward POST requests to doGet
    protected void doPost(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {
        doGet(request, response);
    }
}

package com.example.java;

import jakarta.servlet.ServletConfig;
import jakarta.servlet.ServletContext;
import jakarta.servlet.ServletException;
import jakarta.servlet.annotation.WebServlet;
import jakarta.servlet.http.HttpServlet;
import jakarta.servlet.http.HttpServletRequest;
import jakarta.servlet.http.HttpServletResponse;
import java.io.IOException;
import java.io.PrintWriter;

@WebServlet("/ConfigExampleServlet")
public class ConfigExampleServlet extends HttpServlet {
    private static final long serialVersionUID = 1L;

    protected void doGet(HttpServletRequest req, HttpServletResponse resp)
            throws ServletException, IOException {

        ServletConfig cfg = getServletConfig();
        ServletContext ctx = getServletContext();

        String initVal = cfg.getInitParameter("sampleParam");
        String ctxVal = ctx.getInitParameter("globalParam");

        resp.setContentType("text/plain");
        PrintWriter out = resp.getWriter();
        out.println("Servlet init param: " + initVal);
        out.println("Context param: " + ctxVal);
    }

    protected void doPost(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {
        doGet(request, response);
    }
}
